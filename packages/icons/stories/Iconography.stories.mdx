import { Meta, IconGallery, IconItem } from "@storybook/addon-docs";
import * as icons from "../build/index";
import { config } from "../icons.config";
import Readme from "../README.md";

<Meta title="Components/Icons/Iconography" />

<Readme />

## Icons

<IconGallery>
  {config.map((icon) => {
    const iconName = `${icon
      .split("-")
      .map((part) => `${part.charAt(0).toUpperCase()}${part.slice(1)}`)
      .join("")}Icon`;
    const I = icons[iconName];
    return (
      <IconItem key={icon} name={icon}>
        <I />
      </IconItem>
    );
  })}
</IconGallery>

## Icons

<IconGallery>
  {config.map(({name}) => {
    const iconName = `${name
      .split("-")
      .map((part) => `${part.charAt(0).toUpperCase()}${part.slice(1)}`)
      .join("")}Icon`; 
    const I = icons[iconName];
    
    return (
      <IconItem key={name} name={name}>
        <I />
      </IconItem>
    );
})}

  <IconItem name="linked-in-logo">
    <LinkedInLogo color="black" />
  </IconItem>

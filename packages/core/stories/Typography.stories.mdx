    const size = token[1].value;
    return (
      <>
        <div
          style={{
            fontSize: "14px",
            fontWeight: "700",
            color: "#333",
            textTransform: "uppercase",
          }}
        >
          {token[0]}
        </div>
        <div style={{ color: "rgba(51,51,51,0.8)", fontSize: "14px" }}>
          {token[1].attributes.category}.{token[1].attributes.type}.
          {token[1].attributes.item}
        </div>
        <div style={{ color: "rgba(51,51,51,0.8)", fontSize: "14px" }}>
          {size}
        </div>
        <div
          style={{
            fontSize: size,
            lineHeight: size,
            whiteSpace: "nowrap",
            textOverflow: "ellipsis",
            overflow: "hidden",
            maxWidth: "400px",
          }}
        >
          {sampleText}
        </div>
      </>
    );
  })}

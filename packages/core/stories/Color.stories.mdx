import { Meta, ColorPalette, ColorItem } from "@storybook/addon-docs";
import * as tokens from "../../core/dist/tokens/js/tokens.js";
import * as dark from "../../core/dist/tokens/js/color.dark.js";

<Meta title="Core/Color" />

# Color

## Background

<ColorPalette>
  {Object.entries(tokens.color.background).map((token, index) => {
    const subtitle = token[1].path.join(".");
    return (
      <ColorItem
        title={token[0].toUpperCase()}
        subtitle={subtitle}
        colors={{
          Light: token[1].value,
          Dark: dark.color.background[token[0]].value,
        }}
        key={index}
      />
    );
  })}
</ColorPalette>

## Surface/Border

<ColorPalette>

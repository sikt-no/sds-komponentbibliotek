import { Card } from "@sikt/sds-card";
import { Button, ButtonGroup } from "@sikt/sds-button";
import { NavigateToNextIcon } from "@sikt/sds-icons";

Pakken inneholder komponenterne `<Card />`. Card er en beholder for tekst, bilder og handlinger i sammenheng med et enkelt emne.

### Tekst

Et enkelt eksempel på å ordne teksttyper i et kort.

<Card
  overlineText="Overline"
  headingText="Heading"
  leadText="Lead"
  style={{ maxWidth: "400px" }}
>
  Body
</Card>

### Call to Action

Be brukeren om å gjøre en handling

<Card
  overlineText="Overline"
  headingText="Heading"
  style={{ maxWidth: "400px" }}
  callToAction={
    <ButtonGroup variant="right">
      <Button variant="subtle">Cancel</Button>
      <Button variant="strong" icon={<NavigateToNextIcon />}>
        Action
      </Button>
    </ButtonGroup>
  }
>
  Tekst som presenterer et valg for brukeren
</Card>

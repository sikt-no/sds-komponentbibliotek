import { Meta, ColorPalette, ColorItem } from "@storybook/blocks";
import * as tokens from "../dist/js/tokens.js";
import * as dark from "../dist/js/color.dark.js";

<Meta title="Tokens/Color" />

# Color

## Brand

<ColorPalette>
  {Object.entries(tokens.color.brand).map((token, index) => {
    return Object.entries(token[1]).map((subtoken, subindex) => {
      return (
        <ColorItem
          title={`${token[0]} / ${subtoken[0]}`}
          subtitle={`--sds-${Object.values(subtoken[1].path).join("-")}`}
          colors={{
            "light-mode": subtoken[1].value,
            "dark-mode": Object.entries(
              Object.entries(dark.color.brand)[index][1],
            )[subindex][1].value,
          }}
          key={`${index}-${subindex}`}
        />
      );
    });
  })}
</ColorPalette>

## Interaction

<ColorPalette>
  {Object.entries(tokens.color.interaction).map((token, index) => {
    return Object.entries(token[1]).map((subtoken, subindex) => {
      return Object.entries(subtoken[1]).map((subsubtoken, subsubindex) => {
        return (
          <ColorItem
            title={`${token[0]} / ${subtoken[0]} / ${subsubtoken[0]}`}
            subtitle={`--sds-${Object.values(subsubtoken[1].path).join("-")}`}
            colors={{
              "light-mode": subsubtoken[1].value,
              "dark-mode": Object.entries(
                Object.entries(
                  Object.entries(dark.color.interaction)[index][1],
                )[subindex][1],
              )[subsubindex][1].value,
            }}
            key={`${index}-${subindex}-${subsubindex}`}
          />
        );
      });
    });
  })}
</ColorPalette>

# Typography

## Font Size

<table>

    </tr>
  </thead>

      return (
        <tr key={token[1].name}>

          <td>{`--sds-${Object.values(token[1].path).join("-")}`}</td>
        </tr>
      );
    })}
  </tbody>
</table>

## Editorial Headline

<table>
  <thead>
    <tr>
      <th>Name</th>
      <th>@Mobile</th>
      <th>@Tablet</th>
      <th>@Desktop</th>
      <th>Token</th>
    </tr>
  </thead>
  <tbody>
    {Object.entries(tokens.typography.editorial.headline).map((token) => {
      return Object.entries(token[1]).map((subtoken) => {
        return (
          <tr key={subtoken[1].name}>
            <td>
              {token[0]}-{subtoken[0]}
            </td>
            <td>{subtoken[1].$value}</td>
            <td>
              {tablet.typography.editorial?.headline?.[token[0]]?.[subtoken[0]]
                ?.$value ?? ""}
            </td>
            <td>
              {desktop.typography.editorial?.headline?.[token[0]]?.[subtoken[0]]
                ?.$value ?? ""}
            </td>
            <td>{`--sds-${Object.values(subtoken[1].path).join("-")}`}</td>
          </tr>
        );
      });
    })}
  </tbody>
</table>
